function onGuiDang() {
      const rows = tableRef.value.getSelectionRows();
      if (!rows.length) {
        return ElMessage.warning('请勾选要归档的数据');
      }
      const ids = rows.map((row) => {
        return row.id;
      });
      guiDangDialog.visible = true;
      guiDangDialog.formData.guidangReason = '';
      guiDangDialog.formData.guidangReamrk = '';
      guiDangDialog.formData.ids = ids;
    }
